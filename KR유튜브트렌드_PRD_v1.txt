1. 프로젝트 개요

1.1  배경 및 목적
한국 유튜브의 최신 동향을 한눈에 파악할 수 있는 대시보드를 구축한다. 인기 영상 순위(카테고리별 TOP 10), 채널 구독자 순위, 일간·주간·월간 구독자 증가율, 연령대별 콘텐츠 선호 분석을 제공하며, 1위 영상은 대시보드 내 인라인 재생을 지원한다.

2. 데이터 아키텍처 & API 설계

2.1  YouTube Data API v3 쿼타 전략
⚠️  YouTube API는 하루 기본 10,000유닛이 제공됩니다. search.list는 100유닛/회로 매우 비싸므로 절대 사용하지 않습니다.



2.2  전체 시스템 아키텍처
[ 브라우저 UI — HTML/CSS/JS ]        │  fetch('/api/videos?cat=music&period=daily')        │  fetch('/api/channels')        │  fetch('/api/growth?period=weekly')        ↓ [ FastAPI 백엔드 ]   ├── /api/videos   → Redis 캐시 확인 → YouTube API → 응답   ├── /api/channels → DB 스냅샷 조회 → YouTube API → 저장   ├── /api/growth   → DB 과거값 - 현재값 = 증가량 계산   └── 스케줄러       → 매일 09:00 전체 갱신 + DB 스냅샷 저장        ↓ [ YouTube Data API v3 ]   [ SQLite DB (스냅샷 저장) ]

2.3  구독자 증가율 계산 방식
YouTube API는 현재 구독자 수만 제공합니다. 증가율은 아래 방식으로 계산합니다.
# 매일 09:00 스케줄러 실행 오늘 구독자 수 = channels.list API 호출 DB에 {channel_id, subscriber_count, timestamp} 저장  # 증가량 계산 일간 증가 = 오늘 저장값 - 어제 저장값 주간 증가 = 오늘 저장값 - 7일 전 저장값 월간 증가 = 오늘 저장값 - 30일 전 저장값 증가율(%) = 증가량 / 기준일 구독자 수 × 100

2.4  연령대 매핑 테이블 (KISDI 연구 기반)
💡  YouTube API는 타 채널의 연령대 데이터를 제공하지 않습니다. KISDI·갤럽·나스미디어 연구 데이터를 기반으로 카테고리-연령대 매핑 테이블을 구현합니다.

[youtube_client.py]
from googleapiclient.discovery import build
import json, os, time

CATEGORY_MAP = {
  "all": "",    "music": "10", "game": "20",
  "ent": "24",  "news": "25",  "edu": "27",
  "beauty": "26","food": "26", "travel": "19", "it": "28"
}
CACHE_DIR = "cache/"
CACHE_TTL = 3600  # 1시간

def get_popular_videos(cat="all", period="daily", max_results=10):
    cache_path = f"{CACHE_DIR}{cat}_{period}.json"
    # 캐시 유효 확인
    if os.path.exists(cache_path):
        if time.time() - os.path.getmtime(cache_path) < CACHE_TTL:
            return json.load(open(cache_path))
    
    youtube = build("youtube","v3",developerKey=os.getenv("YOUTUBE_API_KEY"))
    params = dict(part="snippet,statistics",chart="mostPopular",
                  regionCode="KR",maxResults=max_results)
    if CATEGORY_MAP.get(cat):
        params["videoCategoryId"] = CATEGORY_MAP[cat]
    
    res = youtube.videos().list(**params).execute()
    items = [{"id":v["id"],"title":v["snippet"]["title"],
               "channel":v["snippet"]["channelTitle"],
               "views":v["statistics"].get("viewCount","0"),
               "categoryId":v["snippet"]["categoryId"]}
             for v in res.get("items",[])]
    os.makedirs(CACHE_DIR, exist_ok=True)
    json.dump(items, open(cache_path,"w"), ensure_ascii=False)
    return items

[main.py]
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from youtube_client import get_popular_videos

app = FastAPI()
app.add_middleware(CORSMiddleware, allow_origins=["*"], ...)

@app.get("/api/videos")
def videos(cat:str="all", period:str="daily"):
    return get_popular_videos(cat, period)

[channel_tracker.py]
import sqlite3, json
from googleapiclient.discovery import build

CHANNELS_FILE = "config/channels.json"  # [{id, name}] 목록
DB_PATH = "data/snapshots.db"

def init_db():
    conn = sqlite3.connect(DB_PATH)
    conn.execute("""CREATE TABLE IF NOT EXISTS snapshots
      (id INTEGER PRIMARY KEY, channel_id TEXT,
       subscribers INTEGER, views INTEGER, created_at DATE)""")
    conn.commit(); conn.close()

def fetch_and_save_snapshots():
    channels = json.load(open(CHANNELS_FILE))
    ids = [c["id"] for c in channels]
    # 50개 배치 1회 호출 = 1유닛
    youtube = build("youtube","v3",developerKey=...)
    res = youtube.channels().list(
        part="statistics", id=",".join(ids[:50])
    ).execute()
    conn = sqlite3.connect(DB_PATH)
    from datetime import date
    today = str(date.today())
    for item in res["items"]:
        conn.execute("INSERT INTO snapshots VALUES (NULL,?,?,?,?)",
          (item["id"],
           int(item["statistics"]["subscriberCount"]),
           int(item["statistics"]["viewCount"]), today))
    conn.commit(); conn.close()

def get_growth(period="daily"):
    days = {"daily":1,"weekly":7,"monthly":30}[period]
    conn = sqlite3.connect(DB_PATH)
    # 오늘 vs N일 전 비교 쿼리
    rows = conn.execute("""
      SELECT a.channel_id,
             a.subscribers - b.subscribers as gain,
             ROUND((a.subscribers - b.subscribers)*100.0/b.subscribers, 2) as rate
      FROM snapshots a JOIN snapshots b
      ON a.channel_id=b.channel_id
      WHERE a.created_at=date('now')
      AND b.created_at=date('now',?||' days')
      ORDER BY gain DESC LIMIT 10
    """, (f"-{days}",)).fetchall()
    conn.close()
    return [{"channel_id":r[0],"gain":r[1],"rate":r[2]} for r in rows]
